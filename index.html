ES6 classes


class Iphone {
  constructor() {
    this.assembledIn = 'USA';
  }
  set newAssembledIn(value) {
    this.assembledIn = value;
  }
  get aboutPhone() {
    return `Assembled in ${this.assembledIn}.`;
  }
}
            

let iphone = new Iphone();
iphone.aboutPhone // Assembled in USA.
iphone.newAssembledIn = 'China';
iphone.aboutPhone // Assembled in China.

Inheritance

Crockford version


function extend(Child, Parent) {
  var F = function() { };
  F.prototype = Parent.prototype;
  Child.prototype = new F();
  Child.prototype.constructor = Child;
  Child.superclass = Parent.prototype;
}

Inheritance

ES5 version


function extend(Child, Parent) {
  Child.prototype = Object.create(Parent.prototype);
  Child.prototype.constructor = Child;
  Child.superclass = Parent.prototype;
}

  constructor(owner) {
    this.owner = owner;
  }
}

class IphoneX extends Iphone {
  constructor(owner) {
    super(owner);
  }
}
            

let iphone = new Iphone('Apple');
let iphoneX = new IphoneX('Andrew');
console.log(iphone.owner); // Apple
console.log(iphoneX.owner); // Andrew
            

class Iphone {
  constructor (owner = Iphone.getDefaultOwner()) {
    this.owner = owner;
}

  static getDefaultOwner() {
    return 'Apple';
  }
}

class IphoneX extends Iphone {
  constructor() {
    super();
  }
}

iphoneX.owner // Apple
iphoneX.getDefaultOwner() // TypeError
